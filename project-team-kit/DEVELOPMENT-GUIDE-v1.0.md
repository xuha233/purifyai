use rayon::prelude::*;

pub struct Scanner {
    max_files: usize,
    exclude_patterns: Vec<String>,
}

impl Scanner {
    pub fn scan_directory(&self, path: &str) -> Vec<ScanItem> {
        let mut items = Vec::new();

        WalkDir::new(path)
            .into_iter()
            .par_bridge()
            .filter_map(|e| e.ok())
            .take(self.max_files)
            .for_each(|entry| {
                if let Ok(metadata) = entry.metadata() {
                    if metadata.is_file() {
                        items.push(ScanItem {
                            path: entry.path().to_string_lossy().to_string(),
                            size: metadata.len(),
                            modified: metadata.modified()
                                .ok()
                                .map(|t| t.duration_since(std::time::UNIX_EPOCH).unwrap().as_secs()),
                        });
                    }
                }
            });

        items
    }
}
